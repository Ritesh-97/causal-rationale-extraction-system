\section{Problem Statement}

\begin{frame}{Problem Statement}
\begin{itemize}
    \item Contact centers process \textbf{tens to hundreds of thousands} of agent-customer dialogues
    \item Many conversations result in \textbf{business-critical events}:
    \begin{itemize}
        \item Customer escalations
        \item Refund requests
        \item Churn signals
    \end{itemize}
    \item Current systems flag \textbf{that} events occurred, but not \textbf{why}
    \item Need to identify:
    \begin{itemize}
        \item Which dialogue segments triggered events
        \item What conversational patterns lead to outcomes
        \item Temporal patterns that presage events
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Objectives}
\begin{block}{Task 1: Query-Driven Evidence-Based Causal Explanation}
Generate evidence-based causal explanations for natural language queries about business events
\end{block}

\begin{block}{Task 2: Conversational Follow-Up}
Support contextual follow-up conversations with awareness of previous queries and responses
\end{block}
\end{frame}

\begin{frame}{Data Processing Pipeline}
\begin{enumerate}
    \item \textbf{Transcript Loading}: Support JSON, CSV, TXT formats
    \item \textbf{Preprocessing}:
    \begin{itemize}
        \item Speaker normalization
        \item Text cleaning
        \item Turn segmentation
    \end{itemize}
    \item \textbf{Span Extraction}: Sliding windows (5 turns per span)
    \item \textbf{Vector Indexing}: ChromaDB for efficient retrieval
\end{enumerate}
\end{frame}

\begin{frame}{Retrieval System}
\begin{columns}
\column{0.5\textwidth}
\textbf{Two-Stage Retrieval:}
\begin{enumerate}
    \item \textbf{Semantic Search}
    \begin{itemize}
        \item Sentence transformers
        \item Dense vector similarity
        \item Top-k retrieval
    \end{itemize}
    \item \textbf{Reranking}
    \begin{itemize}
        \item Cross-encoder
        \item Query-span relevance
        \item Precision refinement
    \end{itemize}
\end{enumerate}
\column{0.5\textwidth}
\textbf{Event-Specific Retrieval:}
\begin{itemize}
    \item Filter by event type
    \item Extract causal spans
    \item Temporal proximity scoring
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Causal Analysis}
\begin{block}{Pattern Detection}
\begin{itemize}
    \item \textbf{Temporal}: Spans preceding events
    \item \textbf{Sequential}: Consecutive causal relationships
    \item \textbf{Behavioral}: Hesitation, frustration, repetition
    \item \textbf{Event-Specific}: Escalation/refund/churn triggers
\end{itemize}
\end{block}

\begin{block}{Evidence Scoring}
Weighted combination:
\begin{itemize}
    \item Relevance (40\%) + Temporal (30\%) + Pattern (20\%) + Similarity (10\%)
\end{itemize}
\end{block}
\end{frame}

\section{Methodology}

\begin{frame}{Task 1: Query Processing Flow}
\begin{enumerate}
    \item \textbf{Query Parsing}: Extract event type, intent, key terms
    \item \textbf{Retrieval}: Semantic search + reranking
    \item \textbf{Causal Analysis}: Pattern detection + evidence scoring
    \item \textbf{Explanation Generation}: LLM-based synthesis
    \item \textbf{Response Formatting}: Citations + metadata
\end{enumerate}
\end{frame}

\begin{frame}{Task 2: Follow-Up Processing}
\begin{enumerate}
    \item \textbf{Follow-Up Detection}: Identify contextual queries
    \item \textbf{Context Retrieval}: Get conversation history
    \item \textbf{Query Enhancement}: Combine with context
    \item \textbf{Context-Aware Retrieval}: Refine search with context
    \item \textbf{Contextual Generation}: Generate with awareness
    \item \textbf{Context Update}: Maintain conversation state
\end{enumerate}
\end{frame}

\section{Results \& Insights}

\begin{block}{Challenges}
